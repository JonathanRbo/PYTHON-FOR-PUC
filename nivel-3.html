üí°

Estrutura do try/except

  • try: - Tenta executar esse c√≥digo
  • except TipoErro: - Se der esse erro, faz isso
  • else: - Se N√ÉO der erro nenhum (opcional)
  • finally: - SEMPRE executa, d√™ erro ou n√£o (opcional)

import json

class AlunoPUC:
    def __init__(self, nome, curso):
        self.nome = nome
        self.curso = curso
        self.notas = []

    def adicionar_nota(self, materia, nota):
        self.notas.append({"materia": materia, "nota": nota})

    def calcular_media(self):
        if not self.notas:
            return 0
        total = sum(n["nota"] for n in self.notas)
        return total / len(self.notas)

class SistemaPUC:
    def __init__(self):
        self.alunos = []

    def cadastrar_aluno(self):
        nome = input("Nome: ")
        curso = input("Curso: ")
        aluno = AlunoPUC(nome, curso)
        self.alunos.append(aluno)
        print("‚úÖ Aluno cadastrado!")

    def listar_alunos(self):

        print("\n=== ALUNOS DA PUC ===")

            media = aluno.calcular_media()
            print(f"{i}. {aluno.nome} - {aluno.curso} (M√©dia: {media:.2f})")

    def menu(self):
        while True:
            print("\n=== SISTEMA PUC ===")

            print("2 - Listar alunos")
            print("3 - Sair")

            elif opcao == "2":
                self.listar_alunos()
            elif opcao == "3":

# Executar sistema
